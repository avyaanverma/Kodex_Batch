```css
    #box{
        height: 200px;
        width:200px;
        background-color:crimson;
        position: absolute;
        /* animation ko apply karna */
        animation-name: anime;
        animation-duration: 2s;
        animation-delay: 1s;
    }

    /* 
    @keyframes identifier{
        from{}
        to{}
    } 
    
    */

    @keyframes anime{
        from{
            left: 0;
            rotate: 0;
            background-color:red;
            scale: 1;
        }
        to{
            left:70%;
            rotate: 360deg;
            background-color:blue;
            scale: 0.5;
        }
    }

    @keyframes move{
        0%{
            left:0;
            top:0;
        }

        25%{
            left:50%;
            top:0;
        }

        50%{
            left: 50%;
            top:50%;
        }
        75%{
            left: 0;
            top:50%;
        }

        100%{
            left: 0;
            top:0;
        }
    }
```
